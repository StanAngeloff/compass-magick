def fade(factor = nil)
  Compass::Magick::Utils.assert_type 'factor', factor, Sass::Script::Number
  fade_factor = 255 - (255 * Compass::Magick::Utils.value_of(factor, 1.0, 0.5)).to_i
  Effect.new { |pixel| ChunkyPNG::Color.fade(pixel, fade_factor) }
end
