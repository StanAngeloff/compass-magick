def initialize(&process)
      @process = process
      @block   = lambda do |canvas|
        for y in 0...canvas.height do
          for x in 0...canvas.width do
            canvas.set_pixel(x, y, @process.call(canvas.get_pixel(x, y)))
          end
        end
      end
    end
  end
end
